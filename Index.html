# BMI Calculator for Patients # Function to calculate BMI def calculate_bmi(weight, height): bmi = weight / (height ** 2) return bmi # Function to determine BMI category def bmi_category(bmi): if bmi < 18.5: return "Underweight" elif bmi >= 18.5 and bmi < 24.9: return "Normal weight" elif bmi >= 25 and bmi < 29.9: return "Overweight" else: return "Obese" # Collect patient information print("---- BMI Calculator for Patients ----") name = input("Enter patient name: ") age = input("Enter patient age: ") weight = float(input("Enter weight in kilograms: ")) height = float(input("Enter height in meters: ")) # Calculate BMI bmi = calculate_bmi(weight, height) # Determine BMI category category = bmi_category(bmi) # Display results print("\n---- BMI Result ----") print("Patient Name:", name) print("Age:", age) print("BMI:", round(bmi, 2)) print("Category:", category)
